set(CMAKE_SYSTEM_NAME Linux)

find_program(CMAKE_C_COMPILER clang-20)
find_program(CMAKE_CXX_COMPILER clang++-20)

set(
    CMAKE_C_COMPILER
    "${CMAKE_C_COMPILER}"
    CACHE
    STRING
    "C compiler"
    FORCE
)

set(
    CMAKE_CXX_COMPILER
    "${CMAKE_CXX_COMPILER}"
    CACHE
    STRING
    "C++ compiler"
    FORCE
)


set(CMAKE_CXX_FLAGS_INIT "-nostdinc++ -I/usr/lib/llvm-20/include/c++/v1")
set(CMAKE_EXE_LINKER_FLAGS_INIT "-stdlib=libc++ -lc++abi")

set(CMAKE_LINKER_TYPE MOLD)
